---
title: Encore integration
description: Add Tailrace middleware to Encore.ts raw openai-compat endpoints.
mode: guide
---

```ts
import { createTailrace } from "@tailrace/core";
import { tailraceEncore } from "@tailrace/encore";
import { Service } from "encore.dev/service";

export default new Service("api", {
  middlewares: [tailraceEncore(createTailrace(), { agent: "api" })],
});
```

Use `api.raw` for openai-compat proxies. Full write-up: `docs/guides/encore-integration.md`.
